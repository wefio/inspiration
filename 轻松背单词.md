        doubleClickEnabled: true,
        speed: 3000,
        fontSize: 18,
        bgOpacity: 0.7,
        fadeDuration: 400,
        listMode: 'short', // 'short' 或 'long'
        shortListLength: 20 // 短列表长度
    };

    // 加载配置
    let config = {...defaultConfig};
    try {
        const savedConfig = GM_getValue('wordConfig');
        if (savedConfig) {
            config = {...config, ...savedConfig};
        }
    } catch (e) {
        console.log('初次使用，加载默认配置');
    }

    // 状态变量
    let currentWord = null;
    let wordElement = null;
    let wordTimer = null;
    let lastPos = { x: window.innerWidth - 100, y: 50 };
    let animationQueue = 0;
    let tabActive = true;
    let menuIds = [];
    let isMouseOverVideo = false;
    let flashInterval = null;
    let lastVideoCheck = 0;
    let videoCheckInterval = 1000;

    // 列表管理相关变量
    let currentWordList = []; // 当前使用的单词列表
    let currentListIndex = 0; // 当前列表的索引
    let usedWords = new Set(); // 已使用的单词（用于长列表模式）

    // Fisher-Yates 洗牌算法
    function shuffleArray(array) {
        const shuffled = [...array];
        for (let i = shuffled.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [shuffled[i], shuffled[j]] = [shuffled[j], shuffled[i]];
        }
        return shuffled;
    }

    // 开始新列表
    function startNewList() {
        if (config.listMode === 'short') {
            // 短列表模式：随机选择少量单词
            const shuffled = shuffleArray(wordList);
            currentWordList = shuffled.slice(0, config.shortListLength);
            console.log('开始新短列表，单词数量:', currentWordList.length);
        } else {
            // 长列表模式：使用所有单词，但排除已使用的单词
            const availableWords = wordList.filter(word => !usedWords.has(word));

            if (availableWords.length === 0) {
                // 所有单词都已使用，重置
                usedWords.clear();
                currentWordList = shuffleArray(wordList);
                console.log('重置长列表，所有单词已使用一遍');
            } else {
                currentWordList = shuffleArray(availableWords);
            }
        }

        currentListIndex = 0;
    }

    // 获取下一个单词
    function getNextWord() {
        // 如果当前列表已用完，开始新列表
        if (currentListIndex >= currentWordList.length) {
            startNewList();
        }

        const word = currentWordList[currentListIndex];
        currentListIndex++;

        // 在长列表模式下记录已使用的单词
        if (config.listMode === 'long') {
            usedWords.add(word);
        }

        return word;
    }

    // 切换列表模式
    function toggleListMode() {
        config.listMode = config.listMode === 'short' ? 'long' : 'short';
        GM_setValue('wordConfig', config);

        // 重置列表状态
        usedWords.clear();
        startNewList();
        updateMenuCommands();

        // 视觉反馈
        if (wordElement) {
            const originalColor = wordElement.style.backgroundColor;
            wordElement.style.backgroundColor = 'rgba(0,200,255,0.7)';
            setTimeout(() => {
                if (wordElement) {
                    wordElement.style.backgroundColor = originalColor;
                }
            }, 300);
        }
    }

    // 工具函数：检测元素是否为视频或视频子元素
    function isVideoElement(element) {
        return element && (element.tagName === 'VIDEO' || element.closest('video'));
    }

    // 创建单词元素
    function createWordElement() {
        if (wordElement) return wordElement;

        const element = document.createElement('div');
        element.setAttribute('role', 'status');
        element.setAttribute('aria-live', 'polite');

        Object.assign(element.style, {
            position: 'fixed',
            fontSize: `${config.fontSize}px`,
            backgroundColor: `rgba(0,0,0,${config.bgOpacity})`,
            color: 'white',
            padding: '8px 12px',
            borderRadius: '6px',
            zIndex: '2147483646',
            pointerEvents: 'none',
            userSelect: 'none',
            opacity: '0',
            transition: `opacity ${config.fadeDuration}ms ease, background-color 0.3s ease`,
            whiteSpace: 'nowrap',
            textShadow: '0 0 3px #000',
            boxShadow: '0 2px 8px rgba(0,0,0,0.3)'
        });

        // 确保只添加到第一个body元素
        const bodies = document.getElementsByTagName('body');
        if (bodies.length > 0) {
            bodies[0].appendChild(element);
        } else {
            document.body.appendChild(element);
        }

        wordElement = element;
        return element;
    }

    // 移除单词元素
    function removeWordElement() {
        if (wordElement && wordElement.parentNode) {
            wordElement.remove();
            wordElement = null;
        }
    }

    // 橙色闪烁效果
    function startFlashing() {
        if (flashInterval) clearInterval(flashInterval);
        if (!wordElement) return;

        let isOrange = true;
        flashInterval = setInterval(() => {
            if (wordElement) {
                wordElement.style.backgroundColor = isOrange
                    ? 'rgba(255,165,0,0.7)'
                    : 'rgba(255,140,0,0.5)';
                isOrange = !isOrange;
            }
        }, 500);
    }

    function stopFlashing() {
        if (flashInterval) {
            clearInterval(flashInterval);
            flashInterval = null;
        }
        if (wordElement) {
            wordElement.style.backgroundColor = `rgba(0,0,0,${config.bgOpacity})`;
        }
    }

    // 视频检测函数（降低频率）
    function checkVideoUnderMouse(e) {
        const now = Date.now();
        if (now - lastVideoCheck < videoCheckInterval) return;

        lastVideoCheck = now;
        const nowOverVideo = isVideoElement(e.target);

        if (nowOverVideo && !isMouseOverVideo) {
            // 鼠标进入视频区域
            isMouseOverVideo = true;
            clearInterval(wordTimer);
            wordTimer = null;
            startFlashing();
        } else if (!nowOverVideo && isMouseOverVideo) {
            // 鼠标离开视频区域
            isMouseOverVideo = false;
            if (config.isRunning) {
                updateWordDisplay();
                stopFlashing();
            }
        }
    }

    // 鼠标移动事件处理（降低视频检测频率）
    function handleMouseMove(e) {
        // 更新最后位置
        lastPos.x = e.clientX;
        lastPos.y = e.clientY;

        // 降低频率的视频检测
        checkVideoUnderMouse(e);

        // 只有在显示单词时才更新位置（减少DOM操作）
        if (wordElement && config.isRunning && !isMouseOverVideo &&
            parseFloat(wordElement.style.opacity) > 0.5) {
            wordElement.style.left = `${lastPos.x}px`;
            wordElement.style.top = `${lastPos.y - 70}px`;
        }
    }

    // 双击事件处理
    function handleDoubleClick(e) {
        // 如果在视频元素上，不处理双击事件
        if (isVideoElement(e.target)) {
            return;
        }

        if (!config.doubleClickEnabled) {
            return;
        }
        if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA') return;

        toggleMainFunction();
    }

    // 初始化事件监听
    function initEventListeners() {
        document.removeEventListener('dblclick', handleDoubleClick);
        document.removeEventListener('mousemove', handleMouseMove);

        // 添加事件监听器
        if (config.doubleClickEnabled) {
            document.addEventListener('dblclick', handleDoubleClick);
        }
        document.addEventListener('mousemove', handleMouseMove, { passive: true });
    }

    // 主功能开关
    function toggleMainFunction() {
        const wasRunning = config.isRunning;
        config.isRunning = !config.isRunning;
        GM_setValue('wordConfig', config);

        // 如果从暂停状态恢复，且是短列表模式，开始新列表
        if (config.isRunning && !wasRunning && config.listMode === 'short') {
            startNewList();
        }

        updateWordDisplay();
        updateMenuCommands();

        if (wordElement) {
            wordElement.style.backgroundColor = config.isRunning
                ? `rgba(0,0,0,${config.bgOpacity})`
                : 'rgba(255,165,0,0.7)';
        }
    }

    // 双击功能开关
    function toggleDoubleClickFeature() {
        config.doubleClickEnabled = !config.doubleClickEnabled;
        GM_setValue('wordConfig', config);
        initEventListeners();
        updateMenuCommands();

        // 视觉反馈
        if (wordElement) {
            const originalColor = wordElement.style.backgroundColor;
            wordElement.style.backgroundColor = 'rgba(0,255,0,0.7)';
            setTimeout(() => {
                if (wordElement) {
                    wordElement.style.backgroundColor = originalColor;
                }
            }, 300);
        }
    }

    // 更新单词显示
    function updateWordDisplay() {
        clearInterval(wordTimer);
        wordTimer = null;
        stopFlashing();

        // 如果在视频上，不显示单词
        if (isMouseOverVideo) {
            if (wordElement) {
                wordElement.style.display = 'none';
            }
            return;
        }

        if (config.isRunning && tabActive) {
            showWord();
            wordTimer = setInterval(showWord, config.speed);
            if (wordElement) {
                wordElement.style.display = '';
                wordElement.style.backgroundColor = `rgba(0,0,0,${config.bgOpacity})`;
            }
        } else if (wordElement && wordElement.parentNode) {
            wordElement.style.backgroundColor = 'rgba(255,165,0,0.7)';
        }
    }

    // 显示单词
    function showWord() {
        // 在视频上时不显示单词
        if (!config.isRunning || !tabActive || animationQueue > 0 || isMouseOverVideo) {
            return;
        }

        animationQueue++;
        const startTime = Date.now();

        if (wordElement && parseFloat(wordElement.style.opacity) > 0.5) {
            wordElement.style.opacity = '0';
            setTimeout(() => {
                createNewWord(startTime);
            }, config.fadeDuration);
        } else {
            createNewWord(startTime);
        }
    }

    // 创建新单词
    function createNewWord(startTime) {
        // 在视频上时不创建新单词
        if (!config.isRunning || !tabActive || isMouseOverVideo) {
            animationQueue = 0;
            return;
        }

        const element = createWordElement();
        currentWord = getNextWord();
        element.textContent = currentWord;

        // 更新位置到最新鼠标位置
        element.style.left = `${lastPos.x}px`;
        element.style.top = `${lastPos.y - 70}px`;

        // 强制重绘并显示
        void element.offsetWidth;
        element.style.opacity = '0.9';

        const elapsed = Date.now() - startTime;
        const displayTime = Math.max(100, config.speed - config.fadeDuration - elapsed);

        setTimeout(() => {
            if (wordElement === element) {
                element.style.opacity = '0';
                setTimeout(() => {
                    animationQueue = 0;
                }, config.fadeDuration);
            }
        }, displayTime);
    }

    // 更新菜单命令
    function updateMenuCommands() {
        menuIds.forEach(id => GM_unregisterMenuCommand(id));
        menuIds = [];

        menuIds.push(
            GM_registerMenuCommand(
                `${config.isRunning ? '⏸' : '▶'} ${config.isRunning ? '暂停' : '继续'}背单词`,
                toggleMainFunction
            )
        );

        menuIds.push(
            GM_registerMenuCommand(
                `🖱️ 双击快捷操作: ${config.doubleClickEnabled ? 'ON' : 'OFF'}`,
                toggleDoubleClickFeature
            )
        );

        menuIds.push(
            GM_registerMenuCommand(
                `📋 列表模式: ${config.listMode === 'short' ? '短列表' : '长列表'}`,
                toggleListMode
            )
        );

        // 添加手动刷新列表的选项
        menuIds.push(
            GM_registerMenuCommand(
                '🔄 手动刷新单词列表',
                () => {
                    startNewList();
                    if (wordElement) {
                        const originalColor = wordElement.style.backgroundColor;
                        wordElement.style.backgroundColor = 'rgba(150,0,255,0.7)';
                        setTimeout(() => {
                            if (wordElement) {
                                wordElement.style.backgroundColor = originalColor;
                            }
                        }, 300);
                    }
                }
            )
        );
    }

    // 页面可见性变化处理
    function handleVisibilityChange() {
        const wasTabActive = tabActive;
        tabActive = !document.hidden;

        if (document.hidden) {
            clearInterval(wordTimer);
            wordTimer = null;
            stopFlashing();
            if (wordElement) wordElement.style.display = 'none';
        } else {
            // 如果页面从隐藏变为显示，并且处于运行状态，且是短列表模式，开始新列表
            if (!wasTabActive && tabActive && config.isRunning && config.listMode === 'short') {
                startNewList();
            }

            if (config.isRunning && !wordTimer && !isMouseOverVideo) {
                wordTimer = setInterval(showWord, config.speed);
            }
            if (wordElement && config.isRunning && !isMouseOverVideo) {
                wordElement.style.display = '';
            }
        }
    }

    // 清理函数
    function cleanup() {
        clearInterval(wordTimer);
        clearInterval(flashInterval);
        removeWordElement();

        document.removeEventListener('visibilitychange', handleVisibilityChange);
        document.removeEventListener('dblclick', handleDoubleClick);
        document.removeEventListener('mousemove', handleMouseMove);

        menuIds.forEach(id => GM_unregisterMenuCommand(id));
    }

    // 初始化
    function initialize() {
        // 初始化列表
        startNewList();

        // 添加清理钩子
        window.addEventListener('beforeunload', cleanup);

        document.addEventListener('visibilitychange', handleVisibilityChange);
        initEventListeners();
        updateWordDisplay();
        updateMenuCommands();

        GM_addValueChangeListener('wordConfig', (name, oldVal, newVal) => {
            if (JSON.stringify(config) !== JSON.stringify(newVal)) {
                config = {...newVal};
                initEventListeners();
                updateWordDisplay();
                updateMenuCommands();
            }
        });
    }

    // 启动脚本
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', initialize);
    } else {
        initialize();
    }
})();
